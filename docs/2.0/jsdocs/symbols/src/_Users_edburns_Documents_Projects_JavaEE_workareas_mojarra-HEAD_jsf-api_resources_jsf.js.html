 95         var getTransport = function getTransport() {
 96             var methods = [
 97                 function() {
 98                     return new XMLHttpRequest();
 99                 },
100                 function() {
101                     return new ActiveXObject('Msxml2.XMLHTTP');
102                 },
103                 function() {
104                     return new ActiveXObject('Microsoft.XMLHTTP');
105                 }
106             ];
107 
108             var returnVal;
109             for (var i = 0, len = methods.length; i < len; i++) {
110                 try {
111                     returnVal = methods[i]();
112                 } catch(e) {
113                     continue;
114                 }
115                 return returnVal;
116             }
117             throw new Error('Could not create an XHR object.');
118         };

178         var elementReplace = function elementReplace(d, tempTagName, src) {
179             var parent = d.parentNode;
180             var temp = document.createElement(tempTagName);
181             var result = null;
182             temp.id = d.id;
183 
184             // Creating a head element isn't allowed in IE, so we'll disallow it
185             if (d.nodeName.toLowerCase() === "head") {
186                 throw new Error("Attempted to replace a head element");
187             } else {
188                 temp.innerHTML = src;
189             }
190 
191             result = temp;
192             parent.replaceChild(temp, d);
193             return result;
194         };

197         /**
198          * Do update.
199          * @ignore
200          */
201         var doUpdate = function doUpdate(element, formid) {
202             var id, content, markup, str, state;
203             var stateForm;
204 
205             id = element.getAttribute('id');
206             if (id === "javax.faces.ViewState") {
207 
208                 state = element.firstChild;
209 
210                 // Now set the view state from the server into the DOM
211                 // but only for the form that submitted the request.
212 
213                 stateForm = document.getElementById(formid);
214                 if (!stateForm) {
215                     // if the form went away for some reason, we're going to just return silently.
216                     return;
217                 }
218                 var field = stateForm.elements["javax.faces.ViewState"];
219                 if (typeof field == 'undefined') {
220                     field = document.createElement("input");
221                     field.type = "hidden";
222                     field.name = "javax.faces.ViewState";
223                     stateForm.appendChild(field);
224                 }
225                 field.value = state.text || state.data;
226 
227                 return;
228             }
229 
230             // join the CDATA sections in the markup
231             // RELEASE_PENDING are there allowed to be more than one CDATA?
232             markup = '';
233             for (var j = 0; j < element.childNodes.length; j++) {
234                 content = element.childNodes[j];
235                 markup += content.text || content.data;
236             }
237 
238             // RELEASE_PENDING - doc what this does
239             str = markup.replace(new RegExp('(?:<script.*?>)((\n|\r|.)*?)(?:<\/script>)', 'img'), '');
240 
241             var src = str;
242 
243             // If our special render all markup is present..
244             if (id === "javax.faces.ViewRoot") {
245                 // if src contains <html>, trim the <html> and </html>, if present.
246                 //   if src contains <head>
247                 //      extract the contents of <head> and replace current document's
248                 //      <head> with the contents.
249                 //   if src contains <body>
250                 //      extract the contents of <body> and replace the current
251                 //      document's <body> with the contents.
252                 //   if src does not contain <body>
253                 //      replace the current document's <body> with the contents.
254                 var
255                         htmlStartEx = new RegExp("< *html.*>", "gi"),
256                         htmlEndEx = new RegExp("< */ *html.*>", "gi"),
257                         headStartEx = new RegExp("< *head.*>", "gi"),
258                         headEndEx = new RegExp("< */ *head.*>", "gi"),
259                         bodyStartEx = new RegExp("< *body.*>", "gi"),
260                         bodyEndEx = new RegExp("< */ *body.*>", "gi"),
261                         htmlStart, htmlEnd, headStart, headEnd, bodyStart, bodyEnd;
262                 var srcHead = null, srcBody = null;
263                 // find the current document's "body" element
264                 var docBody = document.getElementsByTagName("body")[0];
265                 // if src contains <html>
266                 if (null !== (htmlStart = htmlStartEx.exec(src))) {
267                     // if src contains </html>
268                     if (null !== (htmlEnd = htmlEndEx.exec(src))) {
269                         src = src.substring(htmlStartEx.lastIndex, htmlEnd.index);
270                     } else {
271                         src = src.substring(htmlStartEx.lastIndex);
272                     }
273                 }
274                 // if src contains <head>
275                 if (null !== (headStart = headStartEx.exec(src))) {
276                     // if src contains </head>
277                     if (null !== (headEnd = headEndEx.exec(src))) {
278                         srcHead = src.substring(headStartEx.lastIndex,
279                                 headEnd.index);
280                     } else {
281                         srcHead = src.substring(headStartEx.lastIndex);
282                     }
283                     // find the "head" element
284                     var docHead = document.getElementsByTagName("head")[0];
285                     if (docHead) {
286                         elementReplace(docHead, "head", srcHead);
287                     }
288                 }
289                 // if src contains <body>
290                 if (null !== (bodyStart = bodyStartEx.exec(src))) {
291                     // if src contains </body>
292                     if (null !== (bodyEnd = bodyEndEx.exec(src))) {
293                         srcBody = src.substring(bodyStartEx.lastIndex,
294                                 bodyEnd.index);
295                     } else {
296                         srcBody = src.substring(bodyStartEx.lastIndex);
297                     }
298                     elementReplace(docBody, "body", srcBody);
299                 }
300                 if (!srcBody) {
301                     elementReplace(docBody, "body", src);
302                 }
303 
304             } else {
305                 var d = $(id);
306                 if (!d) {
307                     throw new Error("During update: " + id + " not found");
308                 }
309                 var parent = d.parentNode;
310                 var temp = document.createElement('div');
311                 temp.id = d.id;
312                 // Trim space padding before assigning to innerHTML
313                 temp.innerHTML = str.replace(/^\s+/g,'').replace(/\s+$/g,'');
314 
315                 parent.replaceChild(temp.firstChild, d);
316             }
317         };

384         /**
385          * Ajax Request Queue
386          * @ignore
387          */
388         var Queue = new function Queue() {
389 
390             // Create the internal queue
391             var queue = [];
392 
393 
394             // the amount of space at the front of the queue, initialised to zero
395             var queueSpace = 0;
396 
397             /** Returns the size of this Queue. The size of a Queue is equal to the number
398              * of elements that have been enqueued minus the number of elements that have
399              * been dequeued.
400              * @ignore
401              */
402             this.getSize = function getSize() {
403                 return queue.length - queueSpace;
404             };
405 
406             /** Returns true if this Queue is empty, and false otherwise. A Queue is empty
407              * if the number of elements that have been enqueued equals the number of
408              * elements that have been dequeued.
409              * @ignore
410              */
411             this.isEmpty = function isEmpty() {
412                 return (queue.length === 0);
413             };
414 
415             /** Enqueues the specified element in this Queue.
416              *
417              * @param element - the element to enqueue
418              * @ignore
419              */
420             this.enqueue = function enqueue(element) {
421                 // Queue the request
422                 queue.push(element);
423             };
424 
425 
426             /** Dequeues an element from this Queue. The oldest element in this Queue is
427              * removed and returned. If this Queue is empty then undefined is returned.
428              *
429              * @returns Object The element that was removed from the queue.
430              * @ignore
431              */
432             this.dequeue = function dequeue() {
433                 // initialise the element to return to be undefined
434                 var element = undefined;
435 
436                 // check whether the queue is empty
437                 if (queue.length) {
438                     // fetch the oldest element in the queue
439                     element = queue[queueSpace];
440 
441                     // update the amount of space and check whether a shift should occur
442                     if (++queueSpace * 2 >= queue.length) {
443                         // set the queue equal to the non-empty portion of the queue
444                         queue = queue.slice(queueSpace);
445                         // reset the amount of space at the front of the queue
446                         queueSpace = 0;
447                     }
448                 }
449                 // return the removed element
450                 return element;
451             };
452 
453             /** Returns the oldest element in this Queue. If this Queue is empty then
454              * undefined is returned. This function returns the same value as the dequeue
455              * function, but does not remove the returned element from this Queue.
456              * @ignore
457              */
458             this.getOldestElement = function getOldestElement() {
459                 // initialise the element to return to be undefined
460                 var element = undefined;
461 
462                 // if the queue is not element then fetch the oldest element in the queue
463                 if (queue.length) {
464                     element = queue[queueSpace];
465                 }
466                 // return the oldest element
467                 return element;
468             };
469         }();

472         /**
473          * AjaxEngine handles Ajax implementation details.
474          * @ignore
475          */
476         var AjaxEngine = function AjaxEngine() {
477 
478             var req = {};                  // Request Object
479             req.url = null;                // Request URL
480             req.context = {};              // Context of request and response
481             req.context.source = null;     // Source of this request
482             req.context.onerror = null;    // Error handler for request
483             req.context.onevent = null;    // Event handler for request
484             req.context.formid = null;     // Form that's the context for this request
485             req.xmlReq = null;             // XMLHttpRequest Object
486             req.async = true;              // Default - Asynchronous
487             req.parameters = {};           // Parameters For GET or POST
488             req.queryString = null;        // Encoded Data For GET or POST
489             req.method = null;             // GET or POST
490             req.status = null;             // Response Status Code From Server
491             req.fromQueue = false;         // Indicates if the request was taken off the queue
492             // before being sent.  This prevents the request from
493             // entering the queue redundantly.
494 
495             req.que = Queue;
496 
497             // Get an XMLHttpRequest Handle
498             req.xmlReq = getTransport();
499             if (req.xmlReq === null) {
500                 return null;
501             }
502 
503             // Set up request/response state callbacks
504             /**
505              * @ignore
506              */
507             req.xmlReq.onreadystatechange = function() {
508                 if (req.xmlReq.readyState === 4) {
509                     req.onComplete();
510                 }
511             };
512 
513             /**
514              * This function is called when the request/response interaction
515              * is complete.  If the return status code is successfull,
516              * dequeue all requests from the queue that have completed.  If a
517              * request has been found on the queue that has not been sent,
518              * send the request.
519              * @ignore
520              */
521             req.onComplete = function onComplete() {
522                 if (req.xmlReq.status && (req.xmlReq.status >= 200 && req.xmlReq.status < 300)) {
523                     sendEvent(req.xmlReq, req.context, "complete");
524                     jsf.ajax.response(req.xmlReq, req.context);
525                 } else {
526                     sendEvent(req.xmlReq, req.context, "complete");
527                     sendError(req.xmlReq, req.context, "httpError");
528                 }
529 
530                 // Regardless of whether the request completed successfully (or not),
531                 // dequeue requests that have been completed (readyState 4) and send
532                 // requests that ready to be sent (readyState 0).
533 
534                 var nextReq = req.que.getOldestElement();
535                 if (nextReq === null || typeof nextReq === 'undefined') {
536                     return;
537                 }
538                 while ((typeof nextReq.xmlReq !== 'undefined' && nextReq.xmlReq !== null) &&
539                        nextReq.xmlReq.readyState === 4) {
540                     req.que.dequeue();
541                     nextReq = req.que.getOldestElement();
542                     if (nextReq === null || typeof nextReq === 'undefined') {
543                         break;
544                     }
545                 }
546                 if (nextReq === null || typeof nextReq === 'undefined') {
547                     return;
548                 }
549                 if ((typeof nextReq.xmlReq !== 'undefined' && nextReq.xmlReq !== null) &&
550                     nextReq.xmlReq.readyState === 0) {
551                     nextReq.fromQueue = true;
552                     nextReq.sendRequest();
553                 }
554             };
555 
556             /**
557              * Utility method that accepts additional arguments for the AjaxEngine.
558              * If an argument is passed in that matches an AjaxEngine property, the
559              * argument value becomes the value of the AjaxEngine property.
560              * Arguments that don't match AjaxEngine properties are added as
561              * request parameters.
562              * @ignore
563              */
564             req.setupArguments = function(args) {
565                 for (var i in args) {
566                     if (args.hasOwnProperty(i)) {
567                         if (typeof req[i] === 'undefined') {
568                             req.parameters[i] = args[i];
569                         } else {
570                             req[i] = args[i];
571                         }
572                     }
573                 }
574             };
575 
576             /**
577              * This function does final encoding of parameters, determines the request method
578              * (GET or POST) and sends the request using the specified url.
579              * @ignore
580              */
581             req.sendRequest = function() {
582                 if (req.xmlReq !== null) {
583                     // if there is already a request on the queue waiting to be processed..
584                     // just queue this request
585                     if (!req.que.isEmpty()) {
586                         if (!req.fromQueue) {
587                             req.que.enqueue(req);
588                             return;
589                         }
590                     }
591                     // If the queue is empty, queue up this request and send
592                     if (!req.fromQueue) {
593                         req.que.enqueue(req);
594                     }
595                     // Some logic to get the real request URL
596                     if (req.generateUniqueUrl && req.method == "GET") {
597                         req.parameters["AjaxRequestUniqueId"] = new Date().getTime() + "" + req.requestIndex;
598                     }
599                     var content = null; // For POST requests, to hold query string
600                     for (var i in req.parameters) {
601                         if (req.parameters.hasOwnProperty(i)) {
602                             if (req.queryString.length > 0) {
603                                 req.queryString += "&";
604                             }
605                             req.queryString += encodeURIComponent(i) + "=" + encodeURIComponent(req.parameters[i]);
606                         }
607                     }
608                     if (req.method === "GET") {
609                         if (req.queryString.length > 0) {
610                             req.url += ((req.url.indexOf("?") > -1) ? "&" : "?") + req.queryString;
611                         }
612                     }
613                     req.xmlReq.open(req.method, req.url, req.async);
614                     if (req.method === "POST") {
615                         if (typeof req.xmlReq.setRequestHeader !== 'undefined') {
616                             req.xmlReq.setRequestHeader('Faces-Request', 'partial/ajax');
617                             req.xmlReq.setRequestHeader('Content-type', 'application/x-www-form-urlencoded');
618                         }
619                         content = req.queryString;
620                     }
621                     sendEvent(req.xmlReq, req.context, "begin");
622                     req.xmlReq.send(content);
623                 }
624             };
625 
626             return req;
627         };

629         /**
630          * Error handling callback.
631          * Assumes that the request has completed.
632          * @ignore
633          */
634         var sendError = function sendError(request, context, status, description, serverErrorName, serverErrorMessage) {
635 
636             // Possible errornames:
637             // httpError
638             // emptyResponse
639             // serverError
640             // malformedXML
641 
642             var sent = false;
643             var data = {};  // data payload for function
644             data.type = "error";
645             data.status = status;
646             // RELEASE_PENDING won't work in response
647             data.source = context.source;
648             data.responseCode = request.status;
649             // RELEASE_PENDING pass a copy, not a reference
650             data.responseXML = request.responseXML;
651             // RELEASE_PENDING won't work in response
652             data.responseText = request.responseText;
653 
654             if (description) {
655                 data.description = description;
656             } else if (status == "httpError") {
657                 data.description = "There was an error communicating with the server, status: "+data.responseCode;
658             } else if (status == "serverError") {
659                 data.description = serverErrorMessage;
660             } else if (status == "emptyResponse") {
661                 data.description = "An emply response was received from the server.  Check server error logs.";
662             } else if (status == "malformedXML") {
663                 data.description = "An invalid XML response was received from the server.";
664             }
665 
666             if (status == "serverError") {
667                 data.errorName = serverErrorName;
668                 data.errorMessage = serverErrorMessage;
669             }
670 
671             // If we have a registered callback, send the error to it.
672             if (context.onerror) {
673                 context.onerror.call(null, data);
674                 sent = true;
675             }
676 
677             for (var i in errorListeners) {
678                 if (errorListeners.hasOwnProperty(i)) {
679                     errorListeners[i].call(null, data);
680                     sent = true;
681                 }
682             }
683 
684             if (!sent && jsf.getProjectStage() === "Development") {
685                 if (status == "serverError") {
686                         alert("serverError: " + serverErrorName + " " + serverErrorMessage);
687                 } else {
688                     alert(status + ": "+ data.description);
689                 }
690             }
691         };

693         /**
694          * Event handling callback.
695          * Request is assumed to have completed, except in the case of event = 'begin'.
696          * @ignore
697          */
698         var sendEvent = function sendEvent(request, context, status) {
699 
700             var data = {};
701             data.type = "event";
702             data.status = status;
703             data.source = context.source;
704             if (status !== 'begin') {
705                 data.responseCode = request.status;
706                 // RELEASE_PENDING pass a copy, not a reference
707                 data.responseXML = request.responseXML;
708                 data.responseText = request.responseText;
709             }
710 
711             if (context.onevent) {
712                 context.onevent.call(null, data);
713             }
714 
715             for (var i in eventListeners) {
716                 if (eventListeners.hasOwnProperty(i)) {
717                     eventListeners[i].call(null, data);
718                 }
719             }
720         };

724             /**
725              * Register a callback for error handling.
726              * <p><b>Usage:</b></p>
727              * <pre><code>
728              * jsf.ajax.addOnError(handleError);

1247                 }

1250

1252                 if (xml === null) {
1253                     sendError(request, context, "emptyResponse");
1254                     return;
1255                 }
1256 
1257                 if (xml.firstChild.tagName === "parsererror") {
1258                     sendError(request, context, "malformedXML");
1259                     return;

1273                     window.location = responseType.getAttribute("url");
1274                     return;
1275                 }
1276 
1277 
1278                 if (responseType.nodeName !== "changes") {
1279                     sendError(request, context, "malformedXML", "Top level node must be one of: changes, redirect, error, received: "+responseType.nodeName+" instead.");
1280                     return;
1281                 }
1282 
1283 
1284                 var changes = responseType.childNodes;
1285 
1286                 try {
1287                     for (var i = 0; i < changes.length; i++) {
1288                         switch (changes[i].nodeName) {
1289                             case "update":
1290                                 doUpdate(changes[i], context.formid);
1291                                 break;
1292                             case "delete":
1293                                 doDelete(changes[i]);
1294                                 break;
1295                             case "insert":
1296                                 doInsert(changes[i]);
1297                                 break;
1298                             case "attributes":
1299                                 doAttributes(changes[i]);
1300                                 break;
1301                             case "eval":
1302                                 doEval(changes[i]);

1309                                 return;
1310                         }
1311                     }

1313                     sendError(request, context, "malformedXML", ex.message);
1314                     return;
1315                 }

1348     /**
1349      * <p>Collect and encode state for input controls associated
1350      * with the specified <code>form</code> element.  This will include
1351      * all input controls of type <code>hidden</code>.</p>
1352      * <p><b>Usage:</b></p>
1353      * <pre><code>
1354      * var state = jsf.getViewState(form);
1355      * </pre></code>
1356      *
1357      * @param form The <code>form</code> element whose contained
1358      * <code>input</code> controls will be collected and encoded.
1359      * Only successful controls will be collected and encoded in
1360      * accordance with: <a href="http://www.w3.org/TR/html401/interact/forms.html#h-17.13.2">
1361      * Section 17.13.2 of the HTML Specification</a>.
1362      *
1363      * @returns String The encoded state for the specified form's input controls.
1364      * @function jsf.getViewState
1365      */
1366     jsf.getViewState = function(form) {
1367         if (!form) {
1368             throw new Error("jsf.getViewState:  form must be set");
1369         }
1370         var els = form.elements;
1371         var len = els.length;
1372         var qString = "";
1373         var addField = function(name, value) {
1374             if (qString.length > 0) {
1375                 qString += "&";
1376             }
1377             qString += encodeURIComponent(name) + "=" + encodeURIComponent(value);
1378         };
1379         for (var i = 0; i < len; i++) {
1380             var el = els[i];
1381             if (!el.disabled) {
1382                 switch (el.type) {
1383                     case 'text':
1384                     case 'password':
1385                     case 'hidden':
1386                     case 'textarea':
1387                         addField(el.name, el.value);
1388                         break;
1389                     case 'select-one':
1390                         if (el.selectedIndex >= 0) {
1391                             addField(el.name, el.options[el.selectedIndex].value);
1392                         }
1393                         break;
1394                     case 'select-multiple':
1395                         for (var j = 0; j < el.options.length; j++) {
1396                             if (el.options[j].selected) {
1397                                 addField(el.name, el.options[j].value);
1398                             }
1399                         }
1400                         break;
1401                     case 'checkbox':
1402                     case 'radio':
1403                         addField(el.name, el.checked + "");
1404                         break;
1405                 }
1406             }
1407         }
1408         return qString;
1409     };
